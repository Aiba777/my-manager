const express = require("express");
const app = express();
const PORT = process.env.PORT || 3000;

// Пример middleware для обработки JSON
app.use(express.json());

// Пример маршрута
app.get("/", (req, res) => {
  res.send("Hello from backend!");
});

app.listen(PORT, () => {
  console.log(`Server is running on port ${PORT}`);
});
